24 if (!ctx) {

102inline X509* createSelfSignedCert(EVP_PKEY* pkey, const std::string& commonName, int days = 3650) {
103 X509* x509 = X509_new();
104 if (!x509) return nullptr;
105
106 ASN1_INTEGER_set(X509_get_serialNumber(x509), 1);
107 X509_gmtime_adj(X509_get_notBefore(x509), 0);
108 X509_gmtime_adj(X509_get_notAfter(x509), 60L * 60L * 24L * days);
109
110 X509_set_pubkey(x509, pkey);
111
112 // Subject/Issuer (for CA or self-signed certs)
113 X509_NAME* name = X509_get_subject_name(x509);
114 X509_NAME_add_entry_by_txt(name, "C", MBSTRING_ASC,
115 reinterpret_cast<const unsigned char*>("US"), -1, -1, 0);
116 X509_NAME_add_entry_by_txt(name, "O", MBSTRING_ASC,
117 reinterpret_cast<const unsigned char*>("MyOrg"), -1, -1, 0);
118 X509_NAME_add_entry_by_txt(name, "CN", MBSTRING_ASC,
119 reinterpret_cast<const unsigned char*>(commonName.c_str()), -1, -1, 0);
120
121 // Self-signed (issuer = subject)
122 X509_set_issuer_name(x509, name);
123
124 if (!X509_sign(x509, pkey, EVP_sha256())) {
125 std::cerr << "Failed to sign certificate\n";
126 X509_free(x509);
127 return nullptr;
128 }
129
130 return x509;
131}
